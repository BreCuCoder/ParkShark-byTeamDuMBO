DROP TABLE IF EXISTS "SHARK"."DIVISION";
DROP TABLE SHARK.MEMBERS;
DROP TABLE SHARK.ADDRESSES;
DROP TABLE SHARK.CONTACT_INFOS;
DROP SEQUENCE SHARK.MEMBERS_SEQ;
DROP SEQUENCE SHARK.ADDRESSES_SEQ;
DROP SEQUENCE SHARK.CONTACT_INFOS_SEQ;
DROP TABLE SHARK.DIVISION;
DROP SEQUENCE SHARK.DIVISION_SEQ;

CREATE TABLE "SHARK"."DIVISION"
   ("ID" NUMBER NOT NULL ENABLE,
	"NAME" VARCHAR2(31 CHAR) NOT NULL ENABLE,
	"ORIGINALNAME" VARCHAR2(31 CHAR) NOT NULL ENABLE,
	"DIRECTOR" VARCHAR2(31 CHAR) NOT NULL ENABLE,
	"PARENTDIVISION" NUMBER(5,0),
	 CONSTRAINT "DIVISION_PK" PRIMARY KEY ("ID")
   );

CREATE TABLE ADDRESSES (
    ID INT PRIMARY KEY,
    STREET_NAME VARCHAR2(255) NOT NULL,
    STREET_NUMBER VARCHAR2(255) NOT NULL,
    POSTAL_CODE VARCHAR2(255) NOT NULL,
    LABEL VARCHAR2(255) NOT NULL
);

CREATE SEQUENCE ADDRESSES_SEQ
START WITH 1
INCREMENT BY 1;

CREATE TABLE CONTACT_INFOS (
    ID INT PRIMARY KEY,
    MOBILE_PHONE VARCHAR2(255) NOT NULL,
    FIXED_LINE VARCHAR2(255) NOT NULL,
    EMAIL VARCHAR2(255) NOT NULL
);

CREATE SEQUENCE CONTACT_INFOS_SEQ
START WITH 1
INCREMENT BY 1;

CREATE TABLE MEMBERS (
    ID INT PRIMARY KEY,
    NAME VARCHAR2(255) NOT NULL,
    ADDRESS_ID INT,
    CONTACT_INFO_ID INT,
    FOREIGN KEY (ADDRESS_ID) REFERENCES ADDRESSES(ID),
    FOREIGN KEY (CONTACT_INFO_ID) REFERENCES CONTACT_INFOS(ID)
);

CREATE SEQUENCE MEMBERS_SEQ
START WITH 1
INCREMENT BY 1;
